# Expires headers for older browsers

<IfModule mod_expires.c>
	<FilesMatch ".(bmp|ico|pdf|flv|jpg|jpeg|png|gif|js|css|gz|eot|opentype|otf|ttf|woff|woff2)$">
		ExpiresActive On
		ExpiresDefault "access plus 1 year"
	</FilesMatch>
	<FilesMatch ".(xml|txt|json)$">
		ExpiresActive On
		ExpiresDefault "access plus 2 days"
	</FilesMatch>
	<FilesMatch ".(html|htm)$">
		ExpiresActive On
		ExpiresDefault "access plus 5 minutes"
	</FilesMatch>
</IfModule>

# Headers for newer browsers

<IfModule mod_headers.c>
	Header unset ETag
	FileETag None
	Header set Connection keep-alive
	# 1 year
	<FilesMatch ".(bmp|ico|pdf|flv|jpg|jpeg|png|gif|js|css|gz|eot|opentype|otf|ttf|woff|woff2)$">
		Header set Cache-Control "max-age=31449600, public"
	</FilesMatch>
	# 2 DAYS
	<FilesMatch ".(xml|txt|json)$">
		Header set Cache-Control "max-age=172800, public, must-revalidate"
	</FilesMatch>
	# 1 HOUR
	<FilesMatch ".(html|htm)$">
		Header set Cache-Control "max-age=300, public"
	</FilesMatch>
	<FilesMatch "\.(js|css|xml|json|gz|html|htm)$">
		Header append Vary: Accept-Encoding
	</FilesMatch>
	Header append Vary: User-Agent
</IfModule>
